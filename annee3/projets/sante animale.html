colors = px.colors.qualitative.Set2
for i, col in enumerate(df.select_dtypes(include='number').columns):
    color = colors[i % len(colors)]
    
    # Créer une figure avec deux sous-graphiques
    fig = make_subplots(rows=1, cols=2, subplot_titles=(f"Histogramme de {col}", f"Boxplot de {col}"))
    
    # Histogramme : définir des bins automatiquement selon la plage des valeurs
    bins = np.histogram_bin_edges(df[col].dropna(), bins=70)
    counts, edges = np.histogram(df[col].dropna(), bins=bins)
    bar_trace = go.Bar(
        x=edges[:-1],  # Bords des bins
        y=counts,  # Comptes dans chaque bin
        name=f"Histogramme : {col}",
        marker=dict(color=color)
    )
    fig.add_trace(bar_trace, row=1, col=1)
    
    # Boxplot
    box_trace = go.Box(
        y=df[col], 
        name=f"Boxplot : {col}",
        marker=dict(color=color)
    )
    fig.add_trace(box_trace, row=1, col=2)
    
    # Ajuster l'échelle des axes
    max_y = max(counts)
    fig.update_yaxes(range=[0, max_y * 1.1], row=1, col=1)  # Ajouter 10 % pour l'axe Y
    fig.update_xaxes(title_text=f"Valeurs de {col}", row=1, col=1)  # Nom de l'axe X
    
    # Mise à jour globale
    fig.update_layout(
        title=f"Visualisation pour {col}",
        showlegend=False,
        template="plotly_white"
    )
    fig.show()

from sklearn.mixture import GaussianMixture

# Reshape data for GMM
df['log_Ct'] = np.log1p(df['Ct'])
X_log = df['log_Ct'].dropna().values.reshape(-1, 1)



# Fit a 2-component Gaussian Mixture Model
gmm = GaussianMixture(n_components=2)
gmm.fit(X_log)
means = gmm.means_.flatten()
covariances = np.sqrt(gmm.covariances_).flatten()

# Compute the threshold (intersection of the two distributions)
threshold = np.exp((means[0] * covariances[1] + means[1] * covariances[0]) / (covariances[0] + covariances[1]))
print(f"Threshold found at: {threshold}")

import plotly.graph_objects as go
import numpy as np
import seaborn as sns
from scipy.signal import find_peaks

# KDE plot
kde = sns.kdeplot(df['Ct'].dropna(), bw_adjust=0.5)
x_vals = kde.get_lines()[0].get_xdata()
y_vals = kde.get_lines()[0].get_ydata()

# Find main peaks (modes) for the two normal distributions
peaks, _ = find_peaks(y_vals, prominence=0.02, distance=20)

if len(peaks) >= 2:
    # Identify the single valley (local minimum) between the two peaks
    valleys, _ = find_peaks(-y_vals, prominence=0.025, distance=5)
    
    if len(valleys) > 0:
        valley_idx = valleys[0]  # First valley found (center of unsure zone)
        valley_x = x_vals[valley_idx]  # Valley location

        # Calculate cumulative distribution (CDF) using the trapezoidal rule
        cdf = np.cumsum(y_vals) * (x_vals[1] - x_vals[0])  # Approximation of CDF

        # -- Calculate distribution portions for left and right --
        cdf_left_max = cdf[valley_idx]                     # CDF up to the valley
        cdf_right_portion = cdf[-1] - cdf[valley_idx]       # Portion of CDF from valley to end
        
        # -- Create a rebased array for the right side of the valley --
        cdf_rebased_right = cdf[valley_idx:] - cdf[valley_idx]
        
        # ---------------------------------------------------------
        # Adjust thresholds for ~90% confidence on each distribution
        # ---------------------------------------------------------
        # For the left distribution:
        #   - low_threshold: the point where 90% of the left distribution is below this value.
        low_threshold_value = 0.90 * cdf_left_max
        
        # For the right distribution:
        #   - high_threshold: the point where only 10% of the right distribution is below this value
        #     (so, effectively 90% is above this value).
        high_threshold_value = 0.10 * cdf_right_portion
        
        # Use interpolation on the respective segments
        low_threshold = np.interp(low_threshold_value, cdf[:valley_idx], x_vals[:valley_idx])
        high_threshold = np.interp(high_threshold_value, cdf_rebased_right, x_vals[valley_idx:])

        print(f"Thresholds found: Low = {low_threshold}, High = {high_threshold}")

        # Plot results using Plotly
        fig = go.Figure()

        # Plot KDE line
        fig.add_trace(go.Scatter(x=x_vals, y=y_vals, mode='lines', name='KDE'))

        # Fill areas for the two distributions with different colors
        fig.add_trace(go.Scatter(
            x=x_vals[:valley_idx],
            y=y_vals[:valley_idx],
            fill='tozeroy',
            fillcolor='rgba(0, 100, 255, 0.3)',
            name='First Distribution'
        ))

        fig.add_trace(go.Scatter(
            x=x_vals[valley_idx:],
            y=y_vals[valley_idx:],
            fill='tozeroy',
            fillcolor='rgba(255, 0, 0, 0.3)',
            name='Second Distribution'
        ))

        # Add vertical lines for thresholds
        fig.add_trace(go.Scatter(
            x=[low_threshold, low_threshold],
            y=[0, max(y_vals)],
            mode='lines',
            line=dict(color='green', dash='dash'),
            name='Low Threshold (90%)'
        ))
        fig.add_trace(go.Scatter(
            x=[high_threshold, high_threshold],
            y=[0, max(y_vals)],
            mode='lines',
            line=dict(color='green', dash='dash'),
            name='High Threshold (90%)'
        ))

        # Add vertical lines for peaks and valley
        fig.add_trace(go.Scatter(
            x=[x_vals[peaks[0]], x_vals[peaks[0]]],
            y=[0, max(y_vals)],
            mode='lines',
            line=dict(color='blue', dash='dash'),
            name='Peak 1'
        ))
        fig.add_trace(go.Scatter(
            x=[x_vals[peaks[1]], x_vals[peaks[1]]],
            y=[0, max(y_vals)],
            mode='lines',
            line=dict(color='blue', dash='dash'),
            name='Peak 2'
        ))
        fig.add_trace(go.Scatter(
            x=[valley_x, valley_x],
            y=[0, max(y_vals)],
            mode='lines',
            line=dict(color='red', dash='dash'),
            name='Valley (Unsure Center)'
        ))

        # Set plot layout
        fig.update_layout(
            title="Thresholds for Class Separation with ~90% Confidence",
            xaxis_title="Ct",
            yaxis_title="Density",
            template="plotly_white",
            showlegend=True
        )

        fig.show()
